#include <stdio.h>
int main() {
    // Write C code here
    int n;
    printf("Enter a row");
    scanf("%d",&n);
    
    for(int i=1;i<=n;i++){
        for(int j=1;j<=n; j++){
            if(j<n-i+1){
                printf("0\t");
            }
            else{
                printf("1\t");
            }
        }
        printf("\n");
    }

    return 0;
}
