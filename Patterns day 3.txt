#include<stdio.h>
int main() {
	int n = 5;
	for(int i = 1;i<=n;i++) {
		for(int j = 1;j<=n;j++) {
			if(j<=i) {
				printf("%3c",'*');
			}else {
				printf("%3c",' ');
			}
		}
		printf("\n");
	}
}

      * * *        2         nc = n*2-1 = 9
    * * * * *      3         x = n; //5  x-- 4
  * * * * * * *    4         y = n; //5  y++ 6

Solution - 5
-------------------------------------------------------------
#include<stdio.h>
int main() {
	int n = 5;
	int x = n, y = n;
	for(int i = 1;i<=n;i++) {
		for(int j = 1;j<=n*2-1;j++) {
			if(j<x || j>y) {
				printf("%3c",' ');
			}else {
				printf("%3c",'*');
			}
		}
		x--;
		y++;
		printf("\n");
	}
}

#include<stdio.h>
int main() {
	int n = 5;
	int N = n*2-1;
	int x = n, y = n;
	
	for(int i = 1;i<=N;i++) {
		for(int j = 1;j<=N;j++) {
			if(j<x || j>y) {
				printf("%3c",' ');
			}else {
				if(j==x || j==y)
				printf("%3c",'*');
				else 
				printf("%3c",' ');
			}
		}
		if(i<n) {
		x--;
		y++;
	    }else {
	    	x++;
	    	y--;
		}
		printf("\n");
	}
}
